## 연결리스트란 ?
연결리스트는 선형적인 데이터 구조라는 점에서 배열과 유사하다. <u>하지만 배열과 달리
연결 리스트의 요소들은 특정 메모리 주소나 인덱스에 저장되지 않는다. 오히려 각 요소는 포인터 또는 다음 객체에 대한 링크를 가지는 독립적인
객체에 가깝다.</u>

연결리스트는 각 데이터 요소를 <u>Node</u>라고 부른다. Node는 데이터, 그리고 다음 노드를 가리키는 링크 2가지로 되어있다.

* 연결 리스트의 시작하는 지점을 head라 부른다.
* 마지막 노드는 null을 가르킨다.

## 연결리스트의 장점
* 연결 리스트는 중간에 요소를 삽입하거나 삭제하는 작업이 배열과 비교하여 훨씬 효율적이다.
* 동적으로 메모리를 할당하므로 메모리를 효율적으로 관리할 수 있다. (배열의 경우 미리 크기를 지정하지만 연결리스트는 추가할때마다 메모리를 할당)
## 연결리스트의 단점
* 배열보다 더많은 메모리를 사용한다 (포인터를 담고 있기 때문에)
* 배열은 인덱스를 사용하여 빠르게 접근이 가능하지만, 연결리스트는 처음부터 탐색하기 때문에 느리다.

## 연결리스트 with JavaScript
```js
class LinkedList {
  constructor() {
    this.head = null;
    this.size = 0;
  }

  // 연결리스트가 비어있는지 확인
  isEmpty() {
    return this.size === 0;
  }

  // 연결리스트 길이 
  getSize() {
    return this.size
  }

  // 새 노드 맨 앞에 Node 추가
  prepend(value) {
    const node = new Node(value);
    // 값이 비어있으면 새 노드를 head로 설정
    if(this.isEmpty()){
      this.head = node;
    // 값이 비어있지 않으면 다음 노드를 현재 head로 설정
    } else {
      node.next = this.head
      this.head = node
    }
    this.size ++
  }

  // 새 노드 맨 뒤에 Node 추가
  append(value) {
    const node = new Node(value);
    if(this.isEmpty()){
      this.head = node;
    } else {
      let prev = this.head
      // 마지막 node까지 이동
      while(prev.next) {
        prev = prev.next
      }
      prev.next = node;
    }
    this.size ++
  }

  // 임의의 위치에 새로운 node 추가
  insert(value, index) {
    // 유효한 범위가 아니면 return
    if (index < 0 || index > this.size) {
      return; 
    }
    if (index === 0) {
      this.prepend(value); // 인덱스가 0이면 prepend()
    } else {
      const node = new Node(value); 
      let prev = this.head;
      // 삽입 위치의 이전 노드까지 이동
      for (let i = 0; i < index - 1; i++) {
        prev = prev.next; 
      }
      node.next = prev.next;
      prev.next = node; 
      this.size++;
    }
  }

  // 주어진 위치 노드 삭제하고 값 반환
  removeFrom(index) {
    // 유효한 범위가 아니면 return
    if (index < 0 || index >= this.size) {
      return null; 
    }
    let removedNode;
    // head 노드를 삭제할 때
    if (index === 0) {
      removedNode = this.head; 
      this.head = this.head.next;
    } else {
      let prev = this.head;
      // 삭제 위치의 이전 노드까지 이동
      for (let i = 0; i < index - 1; i++) {
        prev = prev.next; 
      }
      removedNode = prev.next; 
      prev.next = removedNode.next;
    }
    this.size--; // 리스트의 길이 감소
    return removedNode.value; 
  }

  // 주어진 값을 가진 Node를 삭제하고 값 반환
  removeValue(value) {
    if(this.isEmpty()) {
      return null
    }
    // 헤드 노드가 주어진 값과 일치하는 경우
    if(this.head.value === value) {
      this.head = this.head.next
      this.size --
      return value
    } else {
      let prev = this.head
      // 일치하는 노드를 찾을 떄까지 이동
      while(prev.next && prev.next.value !== value) {
        prev = prev.next
      }
      if(prev.next) {
        removeNode = prev.next
        prev.next = removeNode.next
        this.size --
        return value
      }
      return null
    }
  }

    // Node 값 검색
    search(value) {
      // 리스트가 비어있으면 -1
      if (this.isEmpty()) {
        return -1; 
      }
      let i = 0;
      let curr = this.head;
      while (curr) {
        if (curr.value === value) {
          return i; 
        }
        curr = curr.next;
        i++;
      }
      return -1;
    }

  // 연결 리스트를 거꾸로
  reverse() {
    let prev = null;
    let curr = this.head;
    while (curr) {
      const next = curr.next;
      curr.next = prev;
      prev = curr;
      curr = next;
    }
    this.head = prev; 
  }
  
  // 연결리스트 출력
  print() {
    if(this.isEmpty()){
      console.log("리스트가 비어 있습니다.")
    } else {
      let curr = this.head
      let listValues = ''
      while(curr) {
        listValues += `${curr.value} `
        curr = curr.next
      }
      console.log(listValues)
    }
  }
}

class Node {
  constructor(value) {
    this.value = value;
    this.next = null;
  }
}
```
